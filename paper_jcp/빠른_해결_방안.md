## 🔧 **수정 작업 순서**

    '1.0': 'Baseline',
    '2.0': 'Energy Efficiency', 
    '3.0': 'Electric Cracking',
    '4.0': 'Hydrogen Technologies',
    '5.0': 'Heat Pumps',
    '6.0': 'Hybrid Systems'
}

# 기술명 컬럼 추가
df['tech_name'] = df['tech_deployed'].map(tech_mapping)
